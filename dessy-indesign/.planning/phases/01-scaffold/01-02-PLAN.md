**Step 1: Create `src/utils/fileIO.js`** — UXP Storage wrappers:
```javascript
const uxp = require('uxp');
const storage = uxp.storage;

async function openXlsxPicker() {
  const file = await storage.localFileSystem.getFileForOpening({
    allowMultiple: false,
    types: ['xlsx']
  });
  return file; // UXP File Entry object, or null if user cancelled
}

async function readFileAsArrayBuffer(fileEntry) {
  const bytes = await fileEntry.read({ format: storage.formats.binary });
  return bytes; // ArrayBuffer — pass to SheetJS in Phase 2
}

module.exports = { openXlsxPicker, readFileAsArrayBuffer };
```

CRITICAL: `storage.localFileSystem.getFileForOpening()` returns `null` silently if `localFileSystem` permission is missing from manifest.json. Verify manifest has `"localFileSystem": "request"` before testing.

**Step 2: Rewrite `src/ui/App.jsx`** — interactive panel with test buttons:
```jsx
const { h } = require('preact');
const { useState } = require('preact/hooks');
const { openXlsxPicker } = require('../utils/fileIO');

function App() {
  const [status, setStatus] = useState('Ready');

  async function testFilePicker() {
    try {
      const file = await openXlsxPicker();
      if (file) {
        setStatus('Selected: ' + file.name);
      } else {
        setStatus('No file selected (cancelled)');
      }
    } catch (err) {
      console.error('File picker error:', err);
      setStatus('Error: ' + err.message);
    }
  }

  function testInDesignDOM() {
    try {
      const { app } = require('indesign');
      if (app.documents.length === 0) {
        setStatus('No document open');
        return;
      }
      const doc = app.documents.item(0);
      const pageCount = doc.pages.length;
      const firstPageName = doc.pages.item(0).name;
      setStatus('Doc: ' + doc.name + ' (' + pageCount + ' pages, first: ' + firstPageName + ')');
    } catch (err) {
      console.error('InDesign DOM error:', err);
      setStatus('Error: ' + err.message);
    }
  }

  return (
    h('div', { style: { padding: '16px' } },
      h('h2', null, 'Dessy'),
      h('p', null, 'Status: ', status),
      h('sp-button', { onclick: testInDesignDOM }, 'Test InDesign DOM'),
      h('sp-button', { onclick: testFilePicker, style: { marginTop: '8px' } }, 'Test File Picker')
    )
  );
}

module.exports = { default: App };
```

KEY DETAILS:
- `require('indesign')` is called INSIDE the function, not at module top-level — this avoids import errors when the module loads during build/test outside InDesign.
- `app.documents.length` guard prevents crash when no document is open (Pitfall 5).
- Uses `.item(0)` for both `documents` and `pages` collection access — NOT bracket notation.
- `sp-button` uses lowercase `onclick` (not `onClick`) — UXP custom elements may not support React-style camelCase event props. If `onclick` does not fire, the fallback is to use a `ref` with `addEventListener('click', handler)` — but try lowercase first as it's simpler.
- Error handling wraps both test functions with try/catch, logging to console AND showing in panel status.
