**Step 2 — Rewrite `src/services/excelParser.js` as ESM:**

Replace the entire stub with:

```javascript
import XLSX from '../vendor/xlsx.full.min.js';

export async function parseExcel(arrayBuffer) {
  // Defensive wrap — UXP may return binary string instead of ArrayBuffer
  const data = typeof arrayBuffer === 'string'
    ? new TextEncoder().encode(arrayBuffer)
    : new Uint8Array(arrayBuffer);

  const workbook = XLSX.read(data, { type: 'array' });
  const sheetName = workbook.SheetNames[0];
  const sheet = workbook.Sheets[sheetName];
  // header:1 returns array-of-arrays; row 0 is headers
  const rows = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: '' });

  if (rows.length === 0) return { headers: [], rows: [] };

  const headers = rows[0].map(String);
  const dataRows = rows.slice(1).filter(row => row.some(cell => cell !== ''));
  return { headers, rows: dataRows };
}
```

Key details:
- ESM export (matches fileIO.js pattern, not CJS)
- `new Uint8Array(arrayBuffer)` wraps the raw buffer for SheetJS `type: 'array'`
- `typeof arrayBuffer === 'string'` guard handles UXP binary string edge case
- `defval: ''` ensures empty cells get empty string not undefined
- `rows[0].map(String)` coerces header values to strings
- Filter removes rows where every cell is empty string

Create `tests/excelParser.test.js` as a plain Node.js test script (no framework).

The test creates real xlsx buffers using the same SheetJS library (write path) and feeds them to parseExcel (read path). This validates the full round-trip.

```javascript
import XLSX from '../src/vendor/xlsx.full.min.js';
import { parseExcel } from '../src/services/excelParser.js';

let passed = 0;
let failed = 0;

function assert(condition, message) {
  if (condition) { passed++; console.log('  PASS:', message); }
  else { failed++; console.error('  FAIL:', message); }
}

function makeXlsx(aoa) {
  // aoa = array of arrays, row 0 = headers
  const ws = XLSX.utils.aoa_to_sheet(aoa);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
  const buf = XLSX.write(wb, { type: 'array', bookType: 'xlsx' });
  return buf.buffer; // ArrayBuffer
}

async function testBasicParse() {
  console.log('Test: basic parse with headers and rows');
  const buf = makeXlsx([
    ['Name', 'Age', 'City'],
    ['Alice', 30, 'Sofia'],
    ['Bob', 25, 'Plovdiv']
  ]);
  const result = await parseExcel(buf);
  assert(result.headers.length === 3, 'has 3 headers');
  assert(result.headers[0] === 'Name', 'first header is Name');
  assert(result.headers[1] === 'Age', 'second header is Age');
  assert(result.headers[2] === 'City', 'third header is City');
  assert(result.rows.length === 2, 'has 2 data rows');
  assert(result.rows[0][0] === 'Alice', 'first row first cell is Alice');
  assert(result.rows[1][2] === 'Plovdiv', 'second row third cell is Plovdiv');
}

async function testEmptyXlsx() {
  console.log('Test: empty xlsx returns empty arrays');
  const ws = XLSX.utils.aoa_to_sheet([]);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
  const buf = XLSX.write(wb, { type: 'array', bookType: 'xlsx' });
  const result = await parseExcel(buf.buffer);
  assert(result.headers.length === 0, 'empty headers');
  assert(result.rows.length === 0, 'empty rows');
}

async function testHeadersOnly() {
  console.log('Test: xlsx with headers but no data rows');
  const buf = makeXlsx([['Col1', 'Col2', 'Col3']]);
  const result = await parseExcel(buf);
  assert(result.headers.length === 3, 'has 3 headers');
  assert(result.rows.length === 0, 'no data rows');
}

async function testEmptyRowsFiltered() {
  console.log('Test: empty rows are filtered out');
  const buf = makeXlsx([
    ['Name', 'Value'],
    ['Alice', 10],
    ['', ''],
    ['Bob', 20]
  ]);
  const result = await parseExcel(buf);
  assert(result.rows.length === 2, 'empty row filtered — 2 rows remain');
  assert(result.rows[0][0] === 'Alice', 'first row is Alice');
  assert(result.rows[1][0] === 'Bob', 'second row is Bob');
}

async function testNumericHeaders() {
  console.log('Test: numeric header values coerced to strings');
  const buf = makeXlsx([[1, 2, 3], ['a', 'b', 'c']]);
  const result = await parseExcel(buf);
  assert(result.headers[0] === '1', 'numeric header coerced to string');
  assert(typeof result.headers[0] === 'string', 'header type is string');
}

async function run() {
  console.log('--- excelParser tests ---');
  await testBasicParse();
  await testEmptyXlsx();
  await testHeadersOnly();
  await testEmptyRowsFiltered();
  await testNumericHeaders();
  console.log(`\nResults: ${passed} passed, ${failed} failed`);
  if (failed > 0) process.exit(1);
}

run();
```

Run with: `node --experimental-vm-modules tests/excelParser.test.js`
