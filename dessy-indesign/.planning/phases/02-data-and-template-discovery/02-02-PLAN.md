```javascript
// Tag regex used by scanDocument — matches {{WordChars}}
const TAG_REGEX = /\{\{(\w+)\}\}/g;

/**
 * Scan the active InDesign document for {{Tag}} placeholders.
 * Text tags: found via doc.stories (handles threaded text frames).
 * Image tags: found via doc.rectangles .label property.
 * Returns tag names + frame references for navigation and highlighting.
 */
export function scanDocument() {
  const { app } = require('indesign');
  const doc = app.activeDocument;
  const textTags = new Map();  // tagName -> first frame ref
  const imageTags = new Map(); // tagName -> rectangle ref

  // Scan stories (not textFrames) to handle threaded text
  const storyCount = doc.stories.length;
  for (let i = 0; i < storyCount; i++) {
    const story = doc.stories.item(i);
    const contents = story.contents || '';
    TAG_REGEX.lastIndex = 0;
    let match;
    while ((match = TAG_REGEX.exec(contents)) !== null) {
      const tagName = match[1];
      if (!textTags.has(tagName)) {
        // Store first text container for navigation
        const navFrame = story.textContainers.length > 0
          ? story.textContainers.item(0)
          : null;
        textTags.set(tagName, navFrame);
      }
    }
  }

  // Scan rectangles (graphic frames) — tag stored in frame .label
  const rectCount = doc.rectangles.length;
  for (let i = 0; i < rectCount; i++) {
    const rect = doc.rectangles.item(i);
    const label = rect.label || '';
    TAG_REGEX.lastIndex = 0;
    const match = TAG_REGEX.exec(label);
    if (match && !imageTags.has(match[1])) {
      imageTags.set(match[1], rect);
    }
  }

  return {
    textTags: Array.from(textTags.keys()),
    imageTags: Array.from(imageTags.keys()),
    frameRefs: { text: textTags, image: imageTags }
  };
}

/**
 * Navigate the InDesign canvas to show the frame containing a tag.
 * Switches to the frame's parent page and selects it.
 */
export function navigateToFrame(frame) {
  if (!frame) return;
  const { app } = require('indesign');
  try {
    app.activeWindow.activePage = frame.parentPage;
    app.select(frame);
  } catch (err) {
    console.error('Navigation failed:', err);
  }
}

/**
 * Toggle highlight stroke on all tagged frames.
 * Uses built-in [Blue] swatch (works in RGB and CMYK documents).
 * active=true: set strokeWeight=2 and blue color.
 * active=false: set strokeWeight=0 to remove highlight.
 */
export function setHighlight(frameRefs, active) {
  if (!frameRefs) return;
  const { app } = require('indesign');
  const doc = app.activeDocument;

  const allFrames = [
    ...frameRefs.text.values(),
    ...frameRefs.image.values()
  ].filter(Boolean); // filter out null nav frames

  let highlightSwatch = null;
  if (active) {
    try {
      highlightSwatch = doc.swatches.itemByName('[Blue]');
      // Force a property read to verify the swatch exists
      highlightSwatch.name;
    } catch (_) {
      try {
        highlightSwatch = doc.swatches.item(3); // fallback: typically Cyan
      } catch (_2) {
        highlightSwatch = null;
      }
    }
  }

  for (const frame of allFrames) {
    try {
      if (active && highlightSwatch) {
        frame.strokeWeight = 2;
        frame.strokeColor = highlightSwatch;
      } else {
        frame.strokeWeight = 0;
      }
    } catch (err) {
      console.error('Highlight failed for frame:', err);
    }
  }
}
```

```javascript
// Tests for the tag regex extraction logic used by templateScanner
// InDesign DOM functions can only be tested manually in InDesign

const TAG_REGEX = /\{\{(\w+)\}\}/g;

let passed = 0;
let failed = 0;

function assert(condition, message) {
  if (condition) { passed++; console.log('  PASS:', message); }
  else { failed++; console.error('  FAIL:', message); }
}

function extractTags(text) {
  TAG_REGEX.lastIndex = 0;
  const tags = [];
  let match;
  while ((match = TAG_REGEX.exec(text)) !== null) {
    tags.push(match[1]);
  }
  return tags;
}

function testSingleTag() {
  console.log('Test: single tag extraction');
  const tags = extractTags('Hello {{Name}}, welcome!');
  assert(tags.length === 1, 'found 1 tag');
  assert(tags[0] === 'Name', 'tag is Name');
}

function testMultipleTags() {
  console.log('Test: multiple tags in one string');
  const tags = extractTags('Dear {{FirstName}} {{LastName}}, your order {{OrderID}} is ready.');
  assert(tags.length === 3, 'found 3 tags');
  assert(tags[0] === 'FirstName', 'first tag is FirstName');
  assert(tags[1] === 'LastName', 'second tag is LastName');
  assert(tags[2] === 'OrderID', 'third tag is OrderID');
}

function testNoTags() {
  console.log('Test: no tags in plain text');
  const tags = extractTags('This is plain text with no placeholders.');
  assert(tags.length === 0, 'no tags found');
}

function testEmptyString() {
  console.log('Test: empty string');
  const tags = extractTags('');
  assert(tags.length === 0, 'no tags in empty string');
}

function testLabelTag() {
  console.log('Test: tag in label format (image frame convention)');
  const tags = extractTags('{{ProductImage}}');
  assert(tags.length === 1, 'found 1 tag');
  assert(tags[0] === 'ProductImage', 'tag is ProductImage');
}

function testMalformedTags() {
  console.log('Test: malformed tags are not matched');
  const tags1 = extractTags('{Name}');  // single braces
  assert(tags1.length === 0, 'single braces not matched');

  const tags2 = extractTags('{{ Name }}');  // spaces inside — \w+ does not match leading space
  assert(tags2.length === 0, 'spaces inside braces not matched');

  const tags3 = extractTags('{{}}');  // empty tag
  assert(tags3.length === 0, 'empty braces not matched');
}

function testUnderscoreInTag() {
  console.log('Test: underscore and digits in tag name');
  const tags = extractTags('{{product_image_01}}');
  assert(tags.length === 1, 'found 1 tag');
  assert(tags[0] === 'product_image_01', 'tag includes underscore and digits');
}

function testLastIndexReset() {
  console.log('Test: lastIndex reset between calls');
  // First call
  const tags1 = extractTags('{{A}} and {{B}}');
  assert(tags1.length === 2, 'first call finds 2 tags');
  // Second call — would fail without lastIndex reset
  const tags2 = extractTags('{{C}}');
  assert(tags2.length === 1, 'second call finds 1 tag (lastIndex was reset)');
  assert(tags2[0] === 'C', 'second call tag is C');
}

function testMixedContent() {
  console.log('Test: tags mixed with special characters');
  const tags = extractTags('Price: ${{Price}} | Discount: {{Discount}}%');
  assert(tags.length === 2, 'found 2 tags');
  assert(tags[0] === 'Price', 'first tag is Price');
  assert(tags[1] === 'Discount', 'second tag is Discount');
}

console.log('--- templateScanner regex tests ---');
testSingleTag();
testMultipleTags();
testNoTags();
testEmptyString();
testLabelTag();
testMalformedTags();
testUnderscoreInTag();
testLastIndexReset();
testMixedContent();
console.log(`\nResults: ${passed} passed, ${failed} failed`);
if (failed > 0) process.exit(1);
```
