**File 1: `src/ui/DataTable.jsx`**

Compact mini-spreadsheet table per user decisions:
- Headers as column names in thead
- Scrollable horizontally (overflowX: auto) for many columns
- Scrollable vertically (maxHeight ~160px, overflowY: auto) for 5-8 visible rows
- Click on a row to select it
- Selected row gets `#3b5bdb` blue background with white text
- Empty state: "No Excel file loaded." message
- Panel is ~400px wide, so use compact padding (3-6px) and small font (11px)

```jsx
import { h } from 'preact';

export function DataTable({ headers, rows, selectedRow, onSelectRow }) {
  if (!headers || !headers.length) {
    return h('p', {
      style: { fontSize: '12px', color: '#888', padding: '8px' }
    }, 'No Excel file loaded.');
  }

  return h('div', {
    style: { overflowX: 'auto', maxHeight: '160px', overflowY: 'auto', marginBottom: '8px' }
  },
    h('table', {
      style: { borderCollapse: 'collapse', fontSize: '11px', width: '100%' }
    },
      h('thead', null,
        h('tr', null, headers.map((col, i) =>
          h('th', {
            key: i,
            style: {
              padding: '4px 6px',
              borderBottom: '1px solid #444',
              textAlign: 'left',
              whiteSpace: 'nowrap',
              position: 'sticky',
              top: 0,
              background: '#2b2b2b'
            }
          }, col)
        ))
      ),
      h('tbody', null, rows.map((row, rowIdx) =>
        h('tr', {
          key: rowIdx,
          onclick: () => onSelectRow(rowIdx),
          style: {
            background: selectedRow === rowIdx ? '#3b5bdb' : 'transparent',
            color: selectedRow === rowIdx ? '#fff' : 'inherit',
            cursor: 'pointer'
          }
        }, headers.map((_, colIdx) =>
          h('td', {
            key: colIdx,
            style: {
              padding: '3px 6px',
              whiteSpace: 'nowrap',
              maxWidth: '120px',
              overflow: 'hidden',
              textOverflow: 'ellipsis'
            }
          }, row[colIdx] != null ? String(row[colIdx]) : '')
        ))
      ))
    )
  );
}
```

**File 2: `src/ui/TagList.jsx`**

Two-section grouped list per user decisions:
- "Text Tags" section header + list of tag names
- "Image Tags" section header + list of tag names
- Click a tag name to navigate canvas to that frame
- Tag name only shown (no page number or frame info per decision)
- Empty state: helpful message about `{{tag}}` format
- Compact styling to fit in the remaining panel space below the data table

```jsx
import { h } from 'preact';

export function TagList({ textTags, imageTags, onTagClick }) {
  const hasAnyTags = (textTags && textTags.length > 0) || (imageTags && imageTags.length > 0);

  if (!hasAnyTags) {
    return h('div', { style: { padding: '8px', fontSize: '12px', color: '#888' } },
      h('p', null, 'No tags found in the active document.'),
      h('p', { style: { marginTop: '4px', fontSize: '11px' } },
        'Use ', h('code', null, '{{TagName}}'), ' in text frames or frame labels.')
    );
  }

  function renderSection(title, tags, type) {
    if (!tags || tags.length === 0) return null;
    return h('div', { style: { marginBottom: '8px' } },
      h('div', {
        style: {
          fontSize: '11px',
          fontWeight: 'bold',
          color: '#aaa',
          textTransform: 'uppercase',
          letterSpacing: '0.5px',
          marginBottom: '4px'
        }
      }, title + ' (' + tags.length + ')'),
      h('div', null, tags.map(tag =>
        h('div', {
          key: tag,
          onclick: () => onTagClick(tag, type),
          style: {
            padding: '3px 8px',
            fontSize: '12px',
            cursor: 'pointer',
            borderRadius: '3px',
            marginBottom: '2px',
            background: 'transparent'
          },
          onmouseover: (e) => { e.currentTarget.style.background = '#333'; },
          onmouseout: (e) => { e.currentTarget.style.background = 'transparent'; }
        },
          h('code', { style: { fontSize: '11px' } }, '{{' + tag + '}}')
        )
      ))
    );
  }

  return h('div', { style: { padding: '4px 0' } },
    renderSection('Text Tags', textTags, 'text'),
    renderSection('Image Tags', imageTags, 'image')
  );
}
```

**File 3: `src/ui/App.jsx` — Complete rewrite**

Orchestrates all Phase 2 functionality:
- "Load Excel" button triggers openXlsxPicker -> readFileAsArrayBuffer -> parseExcel
- Auto-scan on mount via useEffect (scanDocument on first render if doc is open)
- "Scan Document" button for manual re-scan
- DataTable component with row selection
- TagList component with click-to-navigate
- Toggle button for canvas highlights
- File name displayed above the data table
- All frame references stored in pluginState only (NOT in Preact useState per Pitfall 3)

```jsx
import { h } from 'preact';
import { useState, useEffect } from 'preact/hooks';
import { openXlsxPicker, readFileAsArrayBuffer } from '../utils/fileIO';
import { parseExcel } from '../services/excelParser';
import { scanDocument, navigateToFrame, setHighlight } from '../services/templateScanner';
import { pluginState } from '../state';
import { DataTable } from './DataTable';
import { TagList } from './TagList';

function App() {
  const [fileName, setFileName] = useState(null);
  const [headers, setHeaders] = useState([]);
  const [rows, setRows] = useState([]);
  const [selectedRow, setSelectedRow] = useState(null);
  const [textTags, setTextTags] = useState([]);
  const [imageTags, setImageTags] = useState([]);
  const [highlightOn, setHighlightOn] = useState(false);
  const [status, setStatus] = useState('Ready');

  // Auto-scan on mount
  useEffect(() => {
    try {
      const { app } = require('indesign');
      if (app.documents.length > 0) {
        doScan();
      }
    } catch (err) {
      console.error('Auto-scan skipped:', err);
    }
  }, []);

  async function handleLoadExcel() {
    try {
      setStatus('Picking file...');
      const fileEntry = await openXlsxPicker();
      if (!fileEntry) {
        setStatus('Cancelled');
        return;
      }
      setStatus('Reading file...');
      const buffer = await readFileAsArrayBuffer(fileEntry);
      setStatus('Parsing...');
      const data = await parseExcel(buffer);

      // Update Preact state (serializable only)
      setFileName(fileEntry.name);
      setHeaders(data.headers);
      setRows(data.rows);
      setSelectedRow(null);

      // Update shared pluginState
      pluginState.excelData = data;
      pluginState.selectedRow = null;

      setStatus('Loaded ' + data.rows.length + ' rows');
    } catch (err) {
      console.error('Excel load error:', err);
      setStatus('Error: ' + err.message);
    }
  }

  function handleSelectRow(rowIdx) {
    setSelectedRow(rowIdx);
    pluginState.selectedRow = rowIdx;
  }

  function doScan() {
    try {
      const result = scanDocument();
      setTextTags(result.textTags);
      setImageTags(result.imageTags);
      // Store frame refs in pluginState ONLY (not Preact state)
      pluginState.tags = {
        textTags: result.textTags,
        imageTags: result.imageTags,
        frameRefs: result.frameRefs
      };
      setStatus('Found ' + result.textTags.length + ' text, ' + result.imageTags.length + ' image tags');
    } catch (err) {
      console.error('Scan error:', err);
      setStatus('Scan error: ' + err.message);
    }
  }

  function handleTagClick(tagName, type) {
    const refs = pluginState.tags.frameRefs;
    if (!refs) return;
    const frameMap = type === 'text' ? refs.text : refs.image;
    const frame = frameMap.get(tagName);
    if (frame) navigateToFrame(frame);
  }

  function handleToggleHighlight() {
    const newState = !highlightOn;
    setHighlightOn(newState);
    pluginState.highlightActive = newState;
    const refs = pluginState.tags.frameRefs;
    if (refs) setHighlight(refs, newState);
  }

  return h('div', { style: { padding: '12px', fontSize: '12px' } },
    h('h2', { style: { margin: '0 0 8px 0', fontSize: '16px' } }, 'Dessy'),

    // Status bar
    h('p', { style: { margin: '0 0 8px 0', fontSize: '11px', color: '#888' } }, status),

    // Excel section
    h('div', { style: { marginBottom: '12px' } },
      h('sp-button', { onclick: handleLoadExcel, style: { marginRight: '8px' } }, 'Load Excel'),
      h('sp-button', { onclick: doScan, variant: 'secondary' }, 'Scan Document')
    ),

    // File name
    fileName && h('p', {
      style: { margin: '0 0 4px 0', fontSize: '11px', color: '#aaa' }
    }, fileName),

    // Data table
    h(DataTable, {
      headers: headers,
      rows: rows,
      selectedRow: selectedRow,
      onSelectRow: handleSelectRow
    }),

    // Divider
    h('sp-divider', { style: { margin: '8px 0' } }),

    // Tag list
    h(TagList, {
      textTags: textTags,
      imageTags: imageTags,
      onTagClick: handleTagClick
    }),

    // Highlight toggle
    (textTags.length > 0 || imageTags.length > 0) && h('div', { style: { marginTop: '8px' } },
      h('sp-button', {
        onclick: handleToggleHighlight,
        variant: highlightOn ? 'cta' : 'secondary'
      }, highlightOn ? 'Hide Highlights' : 'Show Highlights')
    )
  );
}

export default App;
```

Key implementation notes:
- `pluginState.tags.frameRefs` holds InDesign DOM objects — NEVER put in useState (Pitfall 3)
- Auto-scan in useEffect with empty deps array `[]` — runs once on mount, NOT on every render
- Guard `app.documents.length > 0` before scan (no-document guard from Phase 1)
- `require('indesign')` inside the useEffect callback, not at module level
- DataTable gets only serializable props (headers, rows, selectedRow, callback)
- TagList gets only tag name strings + callback — frame lookup goes through pluginState
- Highlight toggle stores state in both Preact (for button text) and pluginState (for scanner access)
- `sp-button` and `sp-divider` are Spectrum UXP widgets built into InDesign runtime
- All components use `h()` function calls — consistent with existing App.jsx pattern
