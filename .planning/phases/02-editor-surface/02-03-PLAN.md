All sections share a helper pattern for writing back to canvas:
```typescript
function updateCanvasObject(updates: Record<string, unknown>) {
  const canvas = useCanvasStore.getState().canvasRef;
  const obj = canvas?.getActiveObject();
  if (!obj || !canvas) return;
  obj.set(updates);
  obj.setCoords(); // CRITICAL: always call after position/size changes
  canvas.renderAll();
  // Fire object:modified for history (debounced 300ms by the history system)
  canvas.fire('object:modified', { target: obj });
}
```
