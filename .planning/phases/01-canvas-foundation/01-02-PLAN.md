       import { mmToPx } from '@/lib/units';
       import type { FormatDefinition } from '@/constants/formats';

       export function getCanvasOptions(format: FormatDefinition) {
         const docWidthPx = mmToPx(format.widthMm);
         const docHeightPx = mmToPx(format.heightMm);
         const bleedPx = mmToPx(format.bleedMm);
         return {
           width: docWidthPx + bleedPx * 2,   // canvas includes bleed on all sides
           height: docHeightPx + bleedPx * 2,
           preserveObjectStacking: true,
           fireRightClick: true,
           stopContextMenu: true,
           selection: true,
           backgroundColor: '#FFFFFF',
         };
       }

    2. Create `src/hooks/useElementCreation.ts`:
       - Accept `canvas: Canvas | null`
       - Read `activeTool` from `useCanvasStore`
       - When tool is `text`, `rect`, `circle`, `line`, `image`, or `colorBlock`:
         a. On `mouse:down` (button 0 only, not right-click): record start point via `canvas.getScenePoint(opt.e)`, disable canvas selection, create a semi-transparent preview object
         b. On `mouse:move`: update preview object dimensions from `Math.abs(current - start)`
         c. On `mouse:up`: if dimensions > 5px both axes, commit the object via element-factory, make it selectable, set as active object. If < 5px, remove preview (too small).
         d. After creation, switch tool back to `select` (InDesign behavior)
       - For `text` tool: create a Textbox via `createTextFrame`. After commit, auto-enter editing mode: `textbox.enterEditing()`
       - For `image` tool: create placeholder via `createImageFrame`
       - Skip creation if `canvas.getActiveObject()` exists and user clicked on it (selection, not creation)
       - Handle BOTH creation paths for image: click-drag for empty frame, AND handle drop events for direct image placement (listen for `drop` on the canvas wrapper div, read `FileReader` → `FabricImage.fromURL()` with `crossOrigin: 'anonymous'`)

    3. Create `src/hooks/useCanvasZoomPan.ts`:
       - Accept `canvas: Canvas | null`
       - Mouse wheel zoom: `canvas.on('mouse:wheel', ...)` → `canvas.zoomToPoint(point, newZoom)` → `useCanvasStore.getState().setZoom(newZoom)`
       - Zoom range: 0.1 (10%) to 5 (500%); formula: `zoom *= 0.999 ** delta`
       - Pan in hand tool: `mouse:down` when activeTool is `hand` OR `alt` key is held → set `isDragging`, disable selection

         `mouse:up` → stop dragging, re-enable selection
       - After any zoom/pan: `useCanvasStore.getState().setViewportTransform([...canvas.viewportTransform])`
       - Preset zoom functions: `zoomToFit(canvas)`, `zoomTo(canvas, level)` for 0.5, 1, 2
