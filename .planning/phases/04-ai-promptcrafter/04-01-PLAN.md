    export type PromptStep = 'idle' | 'enriching' | 'customizing' | 'generating' | 'result';

      editorial: string;

      composition: string; // e.g. 'close-up', 'wide shot', 'overhead'
      style: string;       // e.g. 'photorealistic', 'editorial', 'product'
      background: string;  // e.g. 'white', 'bokeh', 'solid color'
    }

    export interface FrameContext {
      frameId: string;
      widthPx: number;
      heightPx: number;
      left: number;
      top: number;
      scaleX: number;
      scaleY: number;
      aspectRatio: string;
      positionHint: string;
      brandColors: string[];
      frameName: string;
    }

    export interface ImageHistoryEntry {
      id: string;

      generatedAt: string;       // ISO timestamp
    }

    **3. Create `src/lib/ai/generate-image.ts`:**

    Export these functions:

    - `enrichPrompt(apiKey: string, baseDescription: string, frameContext: FrameContext): Promise<PromptVariations>` — Calls Gemini text API (gemini-2.5-flash) using the same native fetch pattern as callGemini in generate-leaflet.ts (DO NOT import callGemini — it's not exported; duplicate the fetch pattern). Uses `responseMimeType: 'application/json'`, maxOutputTokens 1024, 120s timeout. Passes `buildEnrichmentSystemPrompt(frameContext)` as system_instruction. Parses JSON response. Wraps in try/catch; on JSON parse failure throw `new Error('Failed to parse enrichment response')`.

    - `callGeminiImage(apiKey: string, prompt: string, aspectRatio?: string): Promise<string>` — Calls `gemini-2.5-flash-image` model at `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image:generateContent`. Sends `responseModalities: ["IMAGE"]` in generationConfig. 120s timeout via AbortController. Extracts `inline_data.data` and `inline_data.mime_type` from response parts. Returns `data:${mime_type};base64,${data}`. On 403, throw error with message mentioning "image generation access" for clear user feedback. On missing inline_data, throw "No image in response".

    - `assemblePrompt(baseVariation: string, customization: PromptCustomization): string` — Joins baseVariation with non-empty customization fields: mood → `"${mood} mood"`, lighting → `"${lighting} lighting"`, composition → as-is, style → `"${style} style"`, background → `"${background} background"`. Join with ", ".

    - `snapAspectRatio(widthPx: number, heightPx: number): string` — Compares widthPx/heightPx to the 5 Gemini ratios (1:1, 3:4, 4:3, 9:16, 16:9). Returns the ratio string of the closest match by absolute difference.

    - `base64ToBlob(dataUrl: string): Blob` — Splits on comma, extracts mime from header, atob decodes, returns Blob.

    - `generateThumbnail(dataUrl: string): Promise<string>` — Uses createImageBitmap + OffscreenCanvas to downscale to 100px wide, returns JPEG data URL at 0.7 quality.

    - `placeImageIntoFrame(canvas: Canvas, frameObj: FabricObject & Record<string, unknown>, dataUrl: string, imageId: string): Promise<void>` — Follows EXACTLY the pattern from ImageSection.tsx: captureUndoState, FabricImage.fromURL(objectUrl) (convert dataUrl to blob then objectUrl first to avoid render blocking per research pitfall 2), scale to match frame bounds, copy custom props (customType, name, id, imageId, fitMode), insertAt + remove + setActiveObject + requestRenderAll. Revoke objectUrl after FabricImage loads.
