- If thumbnail exists: `<img>` with `width: '100%'`, `height: '100%'`, `objectFit: 'cover'`
- Load thumbnail in useEffect: call `getThumbnail(project.id)`, set state. **CRITICAL: cleanup must call `URL.revokeObjectURL(url)` to prevent memory leaks:**
```typescript
useEffect(() => {
  let objectUrl: string | null = null;
  let cancelled = false;
  getThumbnail(project.id).then((url) => {
    if (cancelled) { if (url) URL.revokeObjectURL(url); return; }
    objectUrl = url;
    setThumbUrl(url);
  });
  return () => { cancelled = true; if (objectUrl) URL.revokeObjectURL(objectUrl); };
}, [project.id]);
```
